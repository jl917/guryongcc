#### WrapContainer.jsx 
```js
import React, { Suspense } from 'react';
import Container from './container';

const WrapContainer = (page, isPub = false) => {
  const Page = React.lazy(() => import(`../pages/${page}`));
  const Component = () => (
    <Suspense fallback={<div>loading</div>}>
      <Page />
    </Suspense>
  );
  return () => {
    if (isPub) {
      return <Component />;
    }
    return (
      <Container>
        <Component />
      </Container>
    );
  };
};

export default WrapContainer;
```
