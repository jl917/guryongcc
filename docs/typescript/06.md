# Utils

## Partial

```ts
interface Todo {
  title: string;
  description: string;
}

type Todo2 = Partial<Todo>
// equal
type Todo2 = {
  title?: string | undefined;
  description?: string | undefined;
}
```

## Required

```ts
interface Todo {
  title?: string;
  description?: string;
}

type Todo2 = Required<Todo>
// equal
type Todo2 = {
  title: string;
  description: string;
}
```

## Readonly

```ts
interface Todo {
  title: string;
}

type Todo2 = Readonly<Todo>
// equal
type Todo2 = {
  readonly title: string;
}
```

## Record

```ts
interface Todo {
  title: string;
  description: string;
}

type Name = "dao" | "lang";

type AllUserTodo = Record<Name, Todo>
// equal
type AllUserTodo = {
  dao: Todo,
  lang: Todo,
}
```

## Pick

```ts
interface TodoMeta {
  title: string;
  description: string;
  completed: boolean;
}

type Todo = Pick<TodoMeta, 'title' | 'description'>
// equal
type Todo = {
  title: string;
  description: string;
}
```

## Omit

```ts
interface TodoMeta {
  title: string;
  description: string;
  completed: boolean;
}

type Todo = Omit<TodoMeta, 'completed'>
// equal
type Todo = {
  title: string;
  description: string;
}
```

## Exclude

```ts
type T0 = Exclude<'a' | 'b' | 'c', 'a'>
// equal
type T0 = 'b' | 'c';
```

## Extract

```ts
type T0 = Extract<'a' | 'b' | 'c', 'a' | 'f'>
// equal
type T0 = 'a'
```

## NonNullable

```ts
type T0 = NonNullable<string | number | null | undefined | boolean>
// equal
type T0 = string | number | boolean
```

## Intrinsic String Manipulation Types

```ts
// Uppercase
type T0 = Uppercase<'dao' | 'lang'>
// equal 
type T0 = 'DAO' | 'LANG'

// Lowercase
type T1 = Lowercase<'DAO' | 'LANG'>
// equal 
type T1 = 'dao' | 'lang'

// Capitalize
type T2 = Capitalize<'dao' | 'lang'>
// equal
type T2 = "Dao" | "Lang"

// Uncapitalize
type T3 = Uncapitalize<'Dao' | 'Lang'>
// equal
type T3 = "dao" | "lang"
```
